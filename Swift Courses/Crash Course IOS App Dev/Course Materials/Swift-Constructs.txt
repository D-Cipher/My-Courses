//For Loops
var num_ls = [Int](1...10)

var even_nums: [AnyObject] = []

for number in num_ls {
    
    if number % 2 == 0 {
        even_nums.append(number)
    }
    
}

print(even_nums)

let searchedSong = "Hello"

var songList = ["Hello", "Happy", "Work", "Royals", "Team", "Happy"]

var inList = false

var songNum = 0

for song in songList {
    
    if searchedSong == song {
        inList = true
    }
}

for (index, _) in songList.enumerate() {
    
    if searchedSong == songList[index] {
        
        songNum = index
    }
}

if inList == true {
    print(songNum)
} else {
    print("not in list")
}

songList = Array(Set(songList)) //Sets are like Arrays except they have no ordering and do not allow duplicates.

//While Loops

var currentSong = 0

var songsPlayed: [AnyObject] = []

while currentSong < 10 {
    songsPlayed.append(currentSong)
    currentSong += 1
}

print(songsPlayed)
